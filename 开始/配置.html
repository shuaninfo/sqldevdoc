# 配置

# 目录结构

sqldev ---------------------------------- /opt/sqldev 应用目录
├── config ------------------------------ 配置文件目录
│   ├── __global___.ini
│   ├── app.ini ------------------------- 应用配置
│   ├── gorm.ini ------------------------ 数据库配置
│   └── sqldev.ini ---------------------- 系统配置
├── sqldev ------------------------------ 二进制文件（升级）
├── install.sh -------------------------- 安装脚本
├── init.sh ----------------------------- 初始化脚本
├── start.sh ---------------------------- 启动脚本
└── stop.sh ----------------------------- 关闭脚本

备注：因为使用的开发框架的缘故，有些配置项和应用关系不大，所以不再列出，只显示和应用相关的配置项及说明。

# app.ini 配置说明

#最大查询行数
max_rows       = 10000
#cas登录服务器、
cas_server     = https://cas.login.abc.com
#前端应用服务器地址
ui_server      = http://sqldev.abc.com
#登录模式
login_mode     = normal
#文件导出目录
export_path    = /opt/sqldev/export
export_patch   = 3
export_timeout = 300
log_path       = ./log/
export_batch   = 10
#注册码
license        = xxx
#redis数据库配置
[redis]
pool_size = 10
host      = 127.0.0.1
port      = 6379
password  =

# gorm.ini 配置说明

[default]
name           = app
enable         = true
driver         = mysql
#SQLDEV应用数据库
connstring     = root:root@tcp(127.0.0.1:3306)/sqldev?charset=utf8&parseTime=true&loc=Asia%2FShanghai
max_open_conns = 100
max_idle_conns = 100
#是否打印数据库语句
show_sql       = false

# sqldev.ini 配置说明

#应用端口
addrs                   = 0.0.0.0:9000
